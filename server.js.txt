const express = require("express");
const axios = require("axios");

const app = express();
app.use(express.json());

app.post("/webhook", async (req, res) => {

    const intent = req.body.queryResult.intent.displayName;
    const phone = req.body.originalDetectIntentRequest.payload.data.from;

    if (intent === "send_list") {

        await axios.post("https://api.aisensy.com/v1/message", {
            apiKey: process.env.AISENSY_API_KEY,
            campaignName: "send_list_pdf",
            destination: phone,
            userName: "Customer",
            templateParams: [],
            media: {
                type: "document",
                url: process.env.PDF_URL,
                filename: "Product_List.pdf"
            }
        });
    }

    res.send({
        fulfillmentText: "Sending you the product list now."
    });
});

app.listen(process.env.PORT || 3000);
